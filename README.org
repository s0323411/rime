* terra_pinyin.schema.yaml
    - 可以使用 symbols 表情符號
#+BEGIN_EXAMPLE
punctuator:
  import_preset: symbols # 空兩格
recognizer:
  import_preset: default # 空兩格
  patterns: # 空兩格
    number: "^[-+]?[0-9][.:0-9]*[%]?$" # 空四格
    punct: "^/([0-9]0?|[a-z]+)$" # 空四格
    reverse_lookup: "`[a-z]*'?$" # 空四格
#+END_EXAMPLE
